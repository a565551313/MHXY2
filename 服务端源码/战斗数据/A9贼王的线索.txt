function 战斗准备类:取贼王的线索信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 =取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
          战斗单位.难度 ="中级"
          战斗单位.名称组={"贼王"}
          战斗单位.模型组={"吸血鬼"}
          local 数量=2+取队伍人数(玩家id)
          for i=2,数量 do
              table.insert(战斗单位.名称组,"帮凶")
              table.insert(战斗单位.模型组,"骷髅怪")
          end
          return 战斗单位
end
