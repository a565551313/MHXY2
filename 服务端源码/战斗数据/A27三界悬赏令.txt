function 战斗准备类:取三界悬赏令信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 = 取队伍平均等级(玩家数据[玩家id].队伍,玩家id)+5
          战斗单位.难度 = "中级"
          战斗单位.名称组={"通缉犯"}
          战斗单位.模型组={"强盗"}
          local 数量=2+取队伍人数(玩家id)
          for n=2,数量 do
              local 模型=取随机怪(1,120)
              table.insert(战斗单位.名称组,"帮凶")
              table.insert(战斗单位.模型组,模型[2])
          end 
          return 战斗单位
end
