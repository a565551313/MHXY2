function 战斗准备类:取江湖大盗信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 = 取队伍平均等级(玩家数据[玩家id].队伍,玩家id)-5
          战斗单位.不加修炼 = true
          战斗单位.名称组={"江湖骗子","恍然","蛊惑","色诱","难辨",}
          战斗单位.模型组={"护卫","狂豹人形","幽灵","夜罗刹","龙龟",}
          for i=1,5 do
              战斗单位[i]={饰品=true,变异=true}
          end
          return 战斗单位
end
