function 战斗准备类:取散财童子信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 =取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
          战斗单位.名称组={任务数据[任务id].名称}
          战斗单位.模型组={任务数据[任务id].模型}
          for i=2,6 do
              table.insert(战斗单位.名称组,"恭喜发财")
              table.insert(战斗单位.模型组,"葫芦宝贝")
          end
          return 战斗单位 
end
