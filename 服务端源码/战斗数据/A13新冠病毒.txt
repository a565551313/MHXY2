function 战斗准备类:取冠状病毒信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 = 取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
          战斗单位.系数 = {修炼=1}
          战斗单位.名称组={"冠状病毒王"}
          战斗单位.模型组={"进阶混沌兽"}
          战斗单位[1]={不可封印=true}
          for i=2,10 do
              table.insert(战斗单位.名称组,"病毒精英")
              table.insert(战斗单位.模型组,"进阶混沌兽")
          end
          return 战斗单位
end
