function 战斗准备类:哪吒之哪吒(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 = 取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
          战斗单位.不加修炼 = true
          战斗单位.难度 = "顶级"
          战斗单位.名称组={"哪吒"}
          战斗单位.模型组={"超级红孩儿"}
          战斗单位.加载组={"法术"}
          战斗单位[1]={不可封印=true,饰品=true}
          for i=2,10 do
              table.insert(战斗单位.名称组,"魔化丸童")
              table.insert(战斗单位.模型组,"暗黑童子")
              战斗单位[i]={饰品=true,变异 = true}
          end
          return 战斗单位
end
