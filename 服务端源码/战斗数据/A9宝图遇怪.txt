function 战斗准备类:取宝图遇怪信息(任务id,玩家id)
          local 战斗单位={}
          local 范围={"黑山老妖","地狱战神"}
          local 信息=取等级怪(取随机数(1,80))
          local 模型=取敌人信息(信息[取随机数(1,#信息)])
          战斗单位.等级 = 玩家数据[玩家id].角色.数据.等级+5
          战斗单位.不加修炼 = true
          战斗单位.名称组={"远古魔王","喽罗"}          
          战斗单位.模型组={范围[取随机数(1,#范围)],模型[2]}
          return 战斗单位
end
