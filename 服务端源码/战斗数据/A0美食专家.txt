function 战斗准备类:取美食专家信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 =取队伍平均等级(玩家数据[玩家id].队伍,玩家id)-5
          战斗单位[1]={
              名称=任务数据[任务id].名称,
              模型=任务数据[任务id].模型,
            }
          local 范围={"山贼","赌徒","强盗"}
          for i=2,5 do
              战斗单位[i]={名称="贪吃的食客",模型=范围[取随机数(1,#范围)],变异=true}
          end
          return 战斗单位
end
