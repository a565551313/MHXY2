
function 战斗准备类:取经验宝宝信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 = 取队伍平均等级(玩家数据[玩家id].队伍,玩家id)-5
          战斗单位.不加修炼 = true
          战斗单位.名称组={}
          战斗单位.模型组={}
          for i=1,10 do
              table.insert(战斗单位.名称组,任务数据[任务id].名称)
              table.insert(战斗单位.模型组,任务数据[任务id].模型)
              战斗单位[i]={饰品=true}
          end
          return 战斗单位
end
