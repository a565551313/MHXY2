function 战斗准备类:取星宿任务信息(任务id,玩家id)
          local 战斗单位={}
          战斗单位.等级 =取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
          战斗单位.系数 = {修炼=1}
          战斗单位[1]={
              名称=任务数据[任务id].名称,
              模型="天兵",
              变异=true,
              饰品=true,
          }
          return 战斗单位
end
